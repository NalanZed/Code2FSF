public class AbsSeq_Mutant3 {
    //@ ensures num == 0 ==> \result == 0;
    //@ ensures num != 0 ==> \result == -num;
    public static int Abs(int num) {
        if (num != 0) {
            return -num;
        } else {
            return num;
        }
    }
}
