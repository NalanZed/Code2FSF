public class Conjunction_Mutant2 {
    //@ ensures \result == (b1 ? b2 : true);
    //@ ensures \result <==> (b1 ==> b2);
    public static boolean conjunctOf(boolean b1, boolean b2) {
        if(b1 == false)
            return true;
        if(b2 == false)
            return false;
        return true;
    }
}
