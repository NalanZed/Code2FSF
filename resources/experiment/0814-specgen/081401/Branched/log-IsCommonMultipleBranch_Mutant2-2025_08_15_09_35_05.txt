public class IsCommonMultipleBranch_Mutant2 {
    //@ requires a != 0 && b != 0;
    //@ ensures \result == (m % a != 0 ? true : (m % b != 0 ? false : true));
    public static boolean isCommonMultiple(int a, int b, int m) {
        if (m % a != 0) {
            return true;
        }
        if (m % b != 0) {
            return false;
        }
        return true;
    }
}
