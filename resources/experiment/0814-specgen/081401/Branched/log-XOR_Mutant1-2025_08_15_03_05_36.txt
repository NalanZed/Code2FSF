public class XOR_Mutant1 {
    //@ ensures \result <==> (b1 != b2);
    //@ ensures \result == (b1 ? !b2 : b2);
    public static /*@ pure @*/ boolean xor(boolean b1, boolean b2) {
        if(b1 == true) {
            //@ ensures \result == !b2;
            return !b2;
        }
        else {
            //@ ensures \result == b2;
            return b2;
        }
    }
}
