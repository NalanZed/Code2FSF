public class AbsSeq_Mutant2 {
    //@ requires num != Integer.MIN_VALUE;
    //@ ensures \result >= 0;
    //@ ensures \result == (num < 0 ? -num : num);
    public static int Abs(int num) {
        if (num < 0) {
            return num * -1;
        } else {
            return num;
        }
    }
}
