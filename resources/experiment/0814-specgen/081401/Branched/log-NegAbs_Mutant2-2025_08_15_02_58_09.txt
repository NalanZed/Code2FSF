public class NegAbs_Mutant2 {
    //@ ensures \result == num;
    //@ ensures (num < 0 ==> \result == num) && (num >= 0 ==> \result == num);
    public static int negAbs(int num) {
        if (num < 0)
            return num;
        else
            return num;
    }
}
