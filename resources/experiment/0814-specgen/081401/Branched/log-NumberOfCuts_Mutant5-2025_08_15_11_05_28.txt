public class NumberOfCuts_Mutant5 {

    /*@ requires n >= 1;
      @ ensures n == 1 ==> \result == 0;
      @ ensures n % 2 == 0 && n != 1 ==> \result == n - 1;
      @ ensures n % 2 != 0 && n != 1 ==> \result == n;
      @*/
    public static int numberOfCuts(int n) {
        if (n == 1) {
            return 0;
        }
        if (n % 2 == 0) {
            return n - 1; // mutated line
        }
        return n;
    }
}
