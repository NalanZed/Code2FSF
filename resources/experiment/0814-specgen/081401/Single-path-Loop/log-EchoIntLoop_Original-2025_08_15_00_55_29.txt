public class EchoIntLoop_Original {
    /*@ public normal_behavior
      @   requires x >= 0;
      @   ensures \result == x;
      @   assignable \nothing;
      @*/
    public static int echo(int x) {
        int res = 0;
        //@ maintaining 0 <= i && i <= x;
        //@ maintaining res == i;
        //@ decreases x - i;
        for(int i = 0; i < x; i++) {
            res = res + 1;
        }
        return res;
    }
}
