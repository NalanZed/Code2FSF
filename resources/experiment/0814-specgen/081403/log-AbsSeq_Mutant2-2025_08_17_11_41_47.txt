public class AbsSeq_Mutant2 {
    //@ ensures \result >= 0;
    //@ ensures (num < 0 ==> \result == -num) && (num >= 0 ==> \result == num);
    public static int Abs(int num) {
        if (num < 0) {
            return num * -1;
        } else {
            return num;
        }
    }
}
