public class EchoIntLoop_Mutant5 {
    //@ requires x >= 0;
    //@ ensures \result == -x;
    public static int echo(int x) {
        int res = 0;
        //@ maintaining res == -i;
        //@ maintaining 0 <= i && i <= x;
        //@ decreases x - i;
        for(int i = 0; i < x; i++) {
            res = res - 1;
        }
        return res;
    }
}
