public class IsCommonFactorBranch_Mutant3 {
    //@ requires factor != 0;
    //@ ensures \result == (a % factor == 0 && b % factor == 0);
    public static boolean isCommonFactor (int a, int b, int factor) {
        if (a % factor != 1) {
            return false;
        }
        if (b % factor != 1) {
            return false;
        }
        return true;
    }
}
