public class ChangeCase_Mutant2 {

    //@ requires Character.isLetter(c) || !Character.isLetter(c);
    //@ ensures Character.isLowerCase(c) ==> \result == Character.toUpperCase(c);
    //@ ensures Character.isUpperCase(c) ==> \result == Character.toLowerCase(c);
    //@ ensures !Character.isLetter(c) ==> \result == c;
    public static char changeCase(char c) {
        char result;
        if (c > 'z') {
            result = c;
        } else if (c >= 'a') {
            result = (char) (c - 'a' + 'A');
        } else if (c > 'Z') {
            result = c;
        } else if (c >= 'A') {
            result = (char) (c - 'A' + 'a');
        } else {
            result = c;
        }
        return result;
    }
}
