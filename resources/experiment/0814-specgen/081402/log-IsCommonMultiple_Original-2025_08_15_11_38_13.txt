public class IsCommonMultiple_Original {

    //@ requires a != 0 && b != 0;
    //@ ensures \result == (m % a == 0 && m % b == 0);
    //@ ensures \result <==> (m % a == 0 && m % b == 0);
    //@ ensures \result <==> (\exists int k; k * a == m) && (\exists int l; l * b == m);
    /*@ pure @*/ public static boolean isCommonMultiple(int a, int b, int m) {
        return m % a == 0 && m % b == 0;
    }
}
