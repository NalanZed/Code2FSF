public class SmallestEvenMulBranch_Original {

    /*@ requires n >= 1;
      @ ensures \result % 2 == 0;
      @ ensures n % 2 == 0 ==> \result == n;
      @ ensures n % 2 != 0 ==> \result == 2 * n;
      @ ensures \result >= n;
      @*/
    public static int smallestEvenMultiple(int n) {
        if (n % 2 == 0) {
            return n;
        }
        return 2 * n;
    }
}
