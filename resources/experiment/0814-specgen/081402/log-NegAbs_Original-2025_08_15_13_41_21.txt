public class NegAbs_Original {

    //@ requires true;
    //@ ensures \result <= 0;
    //@ ensures num < 0 ==> \result == num;
    //@ ensures num >= 0 ==> \result == -num;
    public static int negAbs(int num) {
        if (num < 0)
            return num;
        else
            return -num;
    }
}
